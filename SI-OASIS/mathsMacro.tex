% Une macro pour les vecteurs qui donne de meilleurs résultats que \overrightarrow.
\def\vect#1{%
	\vbox{\lineskip=-.04em\baselineskip=0pt
	\halign{##\cr
	\leaders\hbox{$\scriptstyle{-}$\kern-.4em}\hfil$\scriptstyle{\rightarrow}$\cr
	$#1$\cr}}}
